[docs]def get_or_create(session, model, **kwargs):
    """Create model if not existing."""
    instance = model.query.filter_by(**kwargs).first()

    if instance:
        return instance

    instance = model(**kwargs)
    session.add(instance)
    session.commit()
    return instance
